How to deal with State in Class Based Components:

  • Since this is class now, we have to initialise with constructor as we have no hooks in classes.
  • No matter how many states are there, we can only have state as an object.
  • Functions inside the function are now replaced by just number.
  • We set the state by using this.setState( { // new state} )
  • We dont override the previous state, but rather merges with the previous state.
import { Component } from 'react/cjs/react.production.min';
import User from './User';

import classes from './Users.module.css';

const DUMMY_USERS = [
  { id: 'u1', name: 'Max' },
  { id: 'u2', name: 'Manuel' },
  { id: 'u3', name: 'Julie' },
];

class Users extends Component {

  constructor() {
    super();
    this.state = {
      showUsers: true,
    };
  }

  toggleUsersHandler = () => {
    this.setState((currState) => {
      return { showUsers: !currState.showUsers }
    }
    )
  };

  render() {

    const usersList = (
      <ul>
        {DUMMY_USERS.map((user) => (
          <User key={user.id} name={user.name} />
        ))} 
      </ul>
    );

    return (
      <div className={classes.users}>
        <button onClick={this.toggleUsersHandler.bind(this)}>
          {this.state.showUsers ? 'Hide' : 'Show'} Users
        </button>
        {this.state.showUsers && usersList}
      </div>
    );
  }
}

export default Users;
