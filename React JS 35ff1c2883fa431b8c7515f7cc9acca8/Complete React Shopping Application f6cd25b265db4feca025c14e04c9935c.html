  1. Adding items to Cart.

    Using useReducer for achieving this.

    const defaultCartState = {
	items : [],
	totalAmount : 0,
}

const CartReducer = (state, action) => {
	if(action.type === 'ADD'){
		const UpdatedItems = state.items.concat(action.item);
		const newTotalAmount = state.totalAmount + (action.item.price * action.time.amount);
		return {
			items : UpdatedItems,
			totalAmount : newTotalAmount
		};
	}
	return (defaultCartState)
}
const CartProvider = (props) => {
	const [CartState, dispatchCartAction] = useReducer(CartReducer, defaultCartState)
	
	const addItemToCartHandler = (item) => {
		dispatchCartAction({type : 'ADD', item : item});
	}
	
	const removeItemToCartHandler = (id) => {
		dispatchCartAction({type : 'REMOVE', id : id});
	}


	const cartCon = {
		items : CartState.items,
		totalAmount : CartState.totalAmount,
		addItem : addItemToCartHandler,
		removeItem : removeItemToCartHandler
	};

	return (
		<CartContext.Provider value={cartCon}>
			{props.children}
		</CartContext.Provider>
	)
}

    Understand that in the useReducer, we have a different syntax than useState. It returns the state and the dispatch function and takes in a function with (state, action) as inputs and also an initial state.

    Ref prop doesnt work because of it being custom components. Hence using Forward Ref.

    Go to the destination and wrap the return component in React.forwardRef
