  1. App.js
import './App.css';
import GoalInput from './components/GoalInput'
import GoalList from './components/GoalList';
import { useState } from 'react';

const initialGoals = [
	{ title: "I wanna become a pilot", id: 99 },
	{ title: "I wanna ride a horse", id: 100 }
];

function App() {

	const [goals, GoalChangingFunction] = useState(initialGoals);
	

	const addGoalHandler = (newGoalText) => {
		if (newGoalText.length !== 0) {
			const newGoal = { title: newGoalText, id: Math.random().toString() }
			console.log(newGoal);
			GoalChangingFunction([newGoal, ...goals]);
		}
	}

	const deletegoalHandler = (idTobeDeleted) => {
		GoalChangingFunction((prevGoals) => {
			const UpdatedGoals = prevGoals.filter((goal) => goal.id !== idTobeDeleted);
			return UpdatedGoals;
		})
	}

	return (
		<div className="App">
			<GoalInput onAddGoal={addGoalHandler}></GoalInput>
			<GoalList goals={goals} onDeleteGoal = {deletegoalHandler}></GoalList>
		</div>
	);
}

export default App;

  1. GoalInput.js
import React, { useState } from 'react';

const GoalInput = (props) => {

    const [enteredGoal, setGoalfn] = useState('');
    const [place, placeFn] = useState('');

    const textHandler = (event) => {
        console.log(enteredGoal);
        placeFn(event.target.value);
        setGoalfn(event.target.value);
    }

    const submitNewGoalHandler = (event) => {
        event.preventDefault();
        if(enteredGoal.length !== 0){
            props.onAddGoal(enteredGoal);
        }
        placeFn('');
    }

    return (
        <form onSubmit={submitNewGoalHandler}>
            <div className="form-control">
                <label>Course Goal: {' '}</label>
                <input type="text" onChange={textHandler} value={place}/>
            </div>
            <button type="submit">Add Goal</button>
        </form>
    )
}

export default GoalInput;

const GoalList = (props) => {

    const deleteGoalHandler = (event) => {
        console.log(event)
        props.onDeleteGoal(event)
    }

    return (
        (props.goals.length === 0 ? <p>How about adding some Goals!? </p> : props.goals.map((goal) => {
            return (
                <div key={goal.id}>
                    <li>{goal.title}&nbsp;&nbsp;<button onClick={() => { deleteGoalHandler(goal.id) }}>Delete</button></li><br></br>
                </div>
            )
        }))
    );
}

export default GoalList;
