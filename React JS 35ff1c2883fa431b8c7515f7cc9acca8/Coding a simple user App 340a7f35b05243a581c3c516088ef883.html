For the part of the overlay, it is important to understand that how it is working. It is just a piece of code which will be executed when there is an error. Yes, error is now a state for us to work on and with the help of that state, we can now can just give the button a toggle functionality. To understand it clear: 

import React from "react";
import Card from "./Card";
import Button from "./Button";

import classes from './ErrorModal.module.css';

const ErrorModal = (props) => {
	return (
		<div>
			<div className={classes.backdrop} onClick = {props.onClickButton}>
			</div>	
			<Card className = {classes.modal}>
			<header className= {classes.header}>
				<h2>
					{props.title}
				</h2>
			</header>

			<div className={classes.content}>
				<p>
					{props.content}
				</p>
			</div>

			<footer className={classes.actions}>
				<Button onClick = {props.onClickButton}> Okay </Button>
			</footer>
		</Card>
		</div>
		
     )
}

export default ErrorModal;
